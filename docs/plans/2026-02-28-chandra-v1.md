# Chandra v1 Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a complete Go-based personal AI agent runtime with structured memory, event-driven triggers, proactive scheduling, and Discord integration.

**Architecture:** Single-binary daemon (`chandrad`) + CLI (`chandra`) backed by SQLite with sqlite-vec for vector search. Four memory layers (episodic, semantic, intent, identity), multi-provider LLM support (OpenAI + Anthropic), MQTT event bus, and a tool system with capability-based security. See `DESIGN.md` for full architecture.

**Tech Stack:** Go 1.22+, SQLite (CGO), sqlite-vec, TOML config, Discord (discordgo), MQTT (paho + mochi-mqtt), OpenAI/Anthropic SDKs, tiktoken-go, cobra CLI, slog logging.

**Reference:** All interface definitions, schema, and implementation notes are in `DESIGN.md`. This plan references sections by number. Read the relevant DESIGN.md section before starting each phase.

package store

import (
	"database/sql"
	"fmt"
	"os"
	"path/filepath"

	_ "github.com/mattn/go-sqlite3"
	vec "github.com/asg017/sqlite-vec-go-bindings/cgo"
)

func init() {
	vec.Auto()
}

// Store wraps the SQLite database connection with Chandra-specific setup.
type Store struct {
	db *sql.DB
}

// NewDB opens (or creates) the SQLite database at path with WAL mode,
// foreign keys, busy timeout, and sqlite-vec loaded.
func NewDB(path string) (*Store, error) {
	if err := os.MkdirAll(filepath.Dir(path), 0700); err != nil {
		return nil, fmt.Errorf("create db directory: %w", err)
	}

	db, err := sql.Open("sqlite3", path+"?_journal_mode=WAL&_foreign_keys=1&_busy_timeout=5000")
	if err != nil {
		return nil, fmt.Errorf("open database: %w", err)
	}

	if err := db.Ping(); err != nil {
		db.Close()
		return nil, fmt.Errorf("ping database: %w", err)
	}

	// SQLite WAL mode supports concurrent readers but serializes writes.
	// Limit write connections to prevent "database is locked" errors.
	db.SetMaxOpenConns(1)

	return &Store{db: db}, nil
}

// DB returns the underlying *sql.DB for direct queries.
func (s *Store) DB() *sql.DB {
	return s.db
}

// Close closes the database connection.
func (s *Store) Close() error {
	return s.db.Close()
}
```

// serializeFloat32 converts []float32 to little-endian bytes for sqlite-vec.
func serializeFloat32(v []float32) []byte {
	buf := make([]byte, len(v)*4)
	for i, f := range v {
		binary.LittleEndian.PutUint32(buf[i*4:], math.Float32bits(f))
	}
	return buf

package store

import (
	"embed"
	"fmt"

	"github.com/golang-migrate/migrate/v4"
	"github.com/golang-migrate/migrate/v4/database/sqlite3"
	"github.com/golang-migrate/migrate/v4/source/iofs"
)

//go:embed migrations/*.sql
var migrationsFS embed.FS

// Migrate applies all pending database migrations.
func (s *Store) Migrate() error {
	source, err := iofs.New(migrationsFS, "migrations")
	if err != nil {
		return fmt.Errorf("open migrations source: %w", err)
	}

	driver, err := sqlite3.WithInstance(s.db, &sqlite3.Config{})
	if err != nil {
		return fmt.Errorf("create migration driver: %w", err)
	}

	m, err := migrate.NewWithInstance("iofs", source, "sqlite3", driver)
	if err != nil {
		return fmt.Errorf("create migrator: %w", err)
	}

	if err := m.Up(); err != nil && err != migrate.ErrNoChange {
		return fmt.Errorf("apply migrations: %w", err)
	}

	return nil
}
```

package store

import (
	"crypto/rand"
	"sync"
	"time"

	"github.com/oklog/ulid/v2"
)

var (
	entropyMu sync.Mutex
	entropy   = ulid.Monotonic(rand.Reader, 0)
)

// NewID returns a new ULID string. Thread-safe, monotonically increasing.
func NewID() string {
	entropyMu.Lock()
	defer entropyMu.Unlock()
	return ulid.MustNew(ulid.Timestamp(time.Now()), entropy).String()
}
